SELECT maker, model, type
FROM Product
WHERE NOT type='PC' OR type='Printer'; // выведёт всё кроме PC

SELECT maker, model, type
FROM Product
WHERE NOT (type='PC' OR type='Printer'); // выведёт всё кроме PC и принтеров

SELECT maker
FROM Product
WHERE NOT (type='PC' AND maker 'A'); // не выведет PC с мейкером A

SELECT maker, model, type 
FROM Product
WHERE type='PC' AND maker='A' OR maker'B' // выведет PC с мейкером А И так же просто мейкеров b

SELECT maker, model, type 
FROM Product
WHERE type='PC' AND (maker='A' OR maker'B') // выведет PC с мейкером А И так же просто мейкеров b

SELECT * FROM product

SELECT * FROM product
WHERE maker = 'A'
UNION 
SELECT * FROM product 
WHERE maker = 'B'; // выведет табличку где мейкеры А и B

SELECT * FROM product
WHERE (maker = 'A' OR maker = 'B') AND type = 'pc'; // выведет PC с мейкерами А и Б

SELECT ram AS Mb, hd Gb
FROM PC
where cd = '24x'; // переименует таблицу ram в Mb и таблицу hd в Gb

SELECT ram*1024 AS kb, hd Gb
FROM PC 
where cd = '24x'; // переведёт RAM значения в Килабайты ( умножив на 1024 )
